121 void CorruptObjects()
122 {
123 foreach (GameObject obj in corruptibleObjects)
124 {
125 if (obj != null && originalPositions.ContainsKey(obj) && originalRotations.ContainsKey(obj))
126 {
127 Vector3 randomOffset = new Vector3(
128 Random.Range(-moveRange, moveRange),
129 Random.Range(-moveRange, moveRange),
130 Random.Range(-moveRange, moveRange)
131 );
132 Vector3 corruptedPosition = originalPositions[obj] + randomOffset;
133
134 Quaternion randomRotation = Quaternion.Euler(
135 originalRotations[obj].eulerAngles.x + Random.Range(-rotationRange, rotationRange),
136 originalRotations[obj].eulerAngles.y + Random.Range(-rotationRange, rotationRange),
137 originalRotations[obj].eulerAngles.z + Random.Range(-rotationRange, rotationRange)
138 );
139
140 obj.transform.position = Vector3.Lerp(
141 obj.transform.position,
142 corruptedPosition,
143 Time.deltaTime * corruptionSpeed
144 );
145 obj.transform.rotation = Quaternion.Lerp(
146 obj.transform.rotation,
147 randomRotation,
148 Time.deltaTime * corruptionSpeed
149 );
150 }
151 }
